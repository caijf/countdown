var oResult = document.getElementById('result');
var countdown = new CountDown({
  time: 2 * 24 * 60 * 60 * 1000,
  interval: 100,
  onChange(currentTime) {
    var timeData = CountDown.parseTimeData(currentTime);
    oResult.innerText = '距活动结束还剩 ' + timeData.days + ' 天 ' + timeData.hours + ' 小时 ' + timeData.minutes + ' 分 ' + timeData.seconds + '.' + timeData.milliseconds/100 + ' 秒';
  },
  onEnd() {
    oResult.innerText = '倒计时结束！';
  }
});

countdown.start();
